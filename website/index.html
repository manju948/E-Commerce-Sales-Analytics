SQL Query Showcase

Some of the SQL queries used to generate business insights.

Monthly Revenue

SELECT
MONTH(order_date) AS Month,
SUM(quantity * price) AS Revenue
FROM Orders o
JOIN Order_Items oi
ON o.order_id = oi.order_id
GROUP BY MONTH(order_date)
ORDER BY Month;

Top 10 Customers

SELECT TOP 10
customer_name,
SUM(quantity * price) AS TotalSpent
FROM Customers c
JOIN Orders o
ON c.customer_id=o.customer_id
JOIN Order_Items oi
ON o.order_id=oi.order_id
GROUP BY customer_name
ORDER BY TotalSpent DESC;

Best Selling Products

SELECT
product_name,
SUM(quantity) AS TotalSold
FROM Products p
JOIN Order_Items oi
ON p.product_id=oi.product_id
GROUP BY product_name
ORDER BY TotalSold DESC;

Sales By Category

SELECT
category,
SUM(quantity*price) AS Revenue
FROM Products p
JOIN Order_Items oi
ON p.product_id=oi.product_id
GROUP BY category;
